#  Import necessary libraries
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Dataset already loaded as df
print("Dataset Shape:", df.shape)
df.head()
#q1
avg_price_by_class = df.groupby('Class')['Fare'].mean()
print("Average Price by Class:")
print(avg_price_by_class)

# Optional bar plot
avg_price_by_class.plot(kind='bar', color='skyblue', title='Average Fare by Class')
plt.ylabel('Average Fare')
plt.show()
#q2
# Group by Days_left and find average fare
price_by_days = df.groupby('Days_left')['Fare'].mean()

plt.figure(figsize=(10,5))
plt.plot(price_by_days, marker='o')
plt.title('Average Fare vs Advance Booking Days')
plt.xlabel('Days Left Before Journey')
plt.ylabel('Average Fare')
plt.grid(True)
plt.show()
#q3
# Introduce check for missing values
print("Missing Fare values before:", df['Fare'].isna().sum())

# Interpolate missing prices
df['Fare'] = df['Fare'].interpolate(method='linear')

print("Missing Fare values after:", df['Fare'].isna().sum())
#q4
# Extract month from Date_of_journey
df['Date_of_journey'] = pd.to_datetime(df['Date_of_journey'])
df['Month'] = df['Date_of_journey'].dt.month

# Correlation
corr_value = df['Month'].corr(df['Fare'])
print(f"Correlation between Month and Fare: {corr_value:.4f}")

# Optional: visualize
plt.figure(figsize=(8,5))
sns.boxplot(x='Month', y='Fare', data=df)
plt.title('Fare Distribution by Month')
plt.show()
#q5
# Line graph of average fare per day
daily_avg = df.groupby('Date_of_journey')['Fare'].mean()

plt.title('Daily Average Fare Over Time')

plt.grid(True)
plt.show()
